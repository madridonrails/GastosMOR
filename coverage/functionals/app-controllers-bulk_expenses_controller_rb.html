320   def import_expense(row, line)
321     @imported_expense = @current_user.expenses.build
322     @imported_expense.date = GastosgemUtils.parse_date(row['fecha'].strip) rescue nil
323     if @imported_expense.date.nil?
324       @bulk_expense.errors.add_to_base("Línea #{line}. La fecha [#{row['fecha']}] no es válida.")
325       @expense_successful = false
326     end
327     @imported_expense.description = row['concepto'].strip
328     @imported_expense.amount = row['importe'].strip rescue nil
329     if @imported_expense.amount.nil?
330       @bulk_expense.errors.add_to_base("Línea #{line}. El importe [#{row['importe']}] no es válido.")
331       @expense_successful = false
332     end
333     @imported_expense.envelope = row['sobre'].strip
334     @imported_expense.payment_type = PaymentType.find_by_name(row['forma pago'].strip) rescue nil
335     if @imported_expense.payment_type.nil?
336       @bulk_expense.errors.add_to_base("Línea #{line}. No se encuentra la forma de pago [#{row['forma pago']}] en la aplicación.")
337       @expense_successful = false
338     end
339     @imported_expense.expense_type = ExpenseType.find_by_description(row['tipo'].strip, :conditions => ['enterprise_id = ?', @current_user.enterprise_id]) rescue nil
340     if @imported_expense.expense_type.nil?
341       @bulk_expense.errors.add_to_base("Línea #{line}. No se encuentra el tipo de gastos [#{row['tipo']}] en la aplicación.")
342       @expense_successful = false
343     end
344     @imported_expense.project = Project.find_by_name(row['proyecto'].strip, :conditions => ['enterprise_id = ?', @current_user.enterprise_id]) rescue nil
345     if @imported_expense.project.nil?
346       @bulk_expense.errors.add_to_base("Línea #{line}. No se encuentra el proyecto [#{row['proyecto']}] en la aplicación.")
347       @expense_successful = false
348     end
349     @imported_expense.status = ExpenseStatus::PENDING
350     @expense_successful
351   end
