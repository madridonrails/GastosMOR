301   def import_payment(row, line)
302     @imported_payment = @current_user.ordered_payments.build
303     @imported_payment.date= GastosgemUtils.parse_date(row['fecha'].strip) rescue nil
304     if @imported_payment.date.nil?

309     if @imported_payment.amount.nil?

311       @payment_successful = false
312     end
313     @imported_payment.concept= row['concepto'].strip
314     @imported_payment.user = User.find_by_email(row['pagado a'].strip, :conditions => ['enterprise_id = ?', @current_user.enterprise_id]) rescue nil
315     if @imported_payment.user.nil?
