297   def approve
298     begin
299       @expense = Expense.find(params[:id])
300       if @expense.pending?
301         @expense.status = ExpenseStatus::APPROVED
302         @expense.revised_by = @current_user.id
303         @expense.revised_at = DateTime.now
304         flash[:notice] = @expense.save ? 'Gasto aprobado satisfactoriamente' : 'Error al aprobar gasto'
305       else
306         flash[:notice] = 'Error al aprobar el gasto. Debe estar pendiente para poder ser aprobado'
307       end
308     rescue Exception => e
309       logger.error(e)
310       flash[:notice] = 'Se intentó aprobar un gasto inexistente'
311     end 
312     
313     return render(:action => 'approve.rjs') if request.xhr?
314     
315     redirect_to :action => 'list_pending'
316   end
