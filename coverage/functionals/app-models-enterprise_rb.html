234       nif_letter = candidate[-1, 1]

236     elsif candidate.match(/X(\d{7})\D/)

240     nif_addition = nif_digits.to_i % 23
241     return true if nif_letter && nif_digits && 'TRWAGMYFPDXBNJZSQVHLCKE'[nif_addition,1] == nif_letter

243     # CIF validation algorithm taken from:
244     # http://club.telepolis.com/jagar1/Ccif.htm

246     central_digits = str_to_int_array(candidate[1,7])

248     a = [1,3,5].inject(0) { |acum, pos| central_digits[pos] + acum }    
249     b = [0,2,4,6].inject(0) { |acum, pos| acum + add_digits_from_int(central_digits[pos] * 2) }
250 
251     candidate_digit = 10 - ( (a+b) % 10)
252     candidate_letter = "JABCDEFGHI"[candidate_digit,1]
