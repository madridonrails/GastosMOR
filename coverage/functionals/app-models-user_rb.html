150   def balance
151     # Note that ''.to_i and nil.to_i give 0.
152     total_pending  = expenses.select {|e| e.pending? && e.payment_type != PaymentType::ENTERPRISE_CREDIT_CARD }.inject(0) {|sum, e| sum + e.amount.to_i}
153     total_approved = expenses.select {|e| e.approved? && e.payment_type != PaymentType::ENTERPRISE_CREDIT_CARD }.inject(0) {|sum, e| sum + e.amount.to_i}
154     total_payments = payments.inject(0) {|sum, p| sum + p.amount.to_i}    
155     total_payments - (total_pending + total_approved)
