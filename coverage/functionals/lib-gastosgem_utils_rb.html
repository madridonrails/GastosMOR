 49 # the formats admited are:

 55 # Brief explain about the regex:
 56 #^(?:\d{1,3} # 1 to 3 digits 
 57 #(?:([, .])\d{3})? # if there is a separator it must be followed by 3 digits 
 58 #(?:\1\d{3})* # if the is more than two groups the same separtor must but used, it must be followed by 3 digits 
 59 #|(?:\d+)) # more than 3 digit with no comma separator 
 60 #((?!\1)[,.]\d{2})?$ # option cents

 62   def self.format_amount(am)
 63     am = am.to_s
 64     if am =~ /^(?:\d{1,3}(?:([, .])\d{3})?(?:\1\d{3})*|(?:\d+))((?!\1)[,.]\d{1,2})?$/
 65       decimal = $2
 66       am.gsub!(decimal,'') unless decimal.nil?  # quit decimals if exist
 67       decimal= "00" if decimal.nil? #if not exists put zero :)
 68       am.gsub!(/[, .]/,'')        # quit punctuation symbols
 69       decimal.gsub!(/[, .]/,'')   # idem. in decimal part
 70       (2 - decimal.size).times do # puts zeros in decimal part
 71           decimal += "0"
 72       end
 73       return (am + decimal)
 74     else
 75       return nil
 76     end
 77   end
