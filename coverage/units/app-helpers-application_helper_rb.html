 25   # We store money as integers to be able to do exact additions and
 26   # subtractions. When we render those quantities back, we deal with
 27   # integers as strings as much as possible to avoid floating-point
 28   # gotchas. We finally delegate to number_to_currency.
 29   def format_money(number, pos=2, unit='', separator=',', delimiter='.')
 30     return number if number.nil? # this is what number_to_currency returns on nil
 31     number = number.to_i
 32     n = if number.abs < 10**pos
 33       width = number < 0 ? pos+2 : pos+1
 34       sprintf("%0#{width}d", number)
 35     else
 36       number.to_s
 37     end
 38     n.sub!(/(\d+)(\d{#{pos}})$/, "\\1\.\\2")        
 39     number_to_currency(n, {
 40       :unit      => unit,
 41       :separator => separator,
 42       :delimiter => delimiter
 43     })
 44   end
